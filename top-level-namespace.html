def source_wait(name='wait')
  total = ENV.fetch("CARNIVORE_SOURCE_#{name.to_s.upcase}", 1.0).to_f
  if(block_given?)
    elapsed = 0.0
    until(yield || elapsed >= total)
      sleep(0.1)
      elapsed += 0.1
    end
    elapsed
  else
    sleep(total)
    total
  end
end
