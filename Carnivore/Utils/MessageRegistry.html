def push(item)
  @store.push(item)
  if(@store.size > @size)
    @store.shift
  end
  self
end

def sha(thing)
  unless(thing.is_a?(String))
    thing = MultiJson.dump(thing)
  end
  (Digest::SHA512.new << thing).hexdigest
end

def valid?(message)
  checksum = sha(message)
  found = @store.include?(checksum)
  unless(found)
    push(checksum)
  end
  !found
end
