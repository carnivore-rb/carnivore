def initialize(*args)
  base = nil
  if(args.first.is_a?(::Hash))
    base = args.shift
  end
  super *args
  if(base)
    self.replace(base)
  end
end

def retrieve(*args)
  args.inject(self) do |memo, key|
    if(memo.is_a?(Hash))
      memo.to_smash[key]
    else
      nil
    end
  end
end

def set(*args)
  unless(args.size > 1)
    raise ArgumentError.new 'Set requires at least one key and a value'
  end
  value = args.pop
  set_key = args.pop
  leaf = args.inject(self) do |memo, key|
    unless(memo[key].is_a?(Hash))
      memo[key] = Smash.new
    end
    memo[key]
  end
  leaf[set_key] = value
  value
end
