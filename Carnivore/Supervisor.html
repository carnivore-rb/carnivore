def terminate!
  if(supervisor)
    begin
      supervisor.terminate
    rescue Celluloid::DeadActorError => e
      Celluloid::Logger.warn "Default supervisor is already in dead state (#{e.class}: #{e})"
    end
    @supervisor = nil
    @registry = nil
  end
  true
end
